// public/script.js - مخصوص GitHub Pages // آدرس Worker خود را در اینجا بگذارید const WORKER_URL = "https://tts-gemini.motanjf.workers.dev"; // وقتی کاربر فرم را ارسال می‌کند document.getElementById('podcastForm').addEventListener('submit', async (e) => { e.preventDefault(); // گرفتن اطلاعات از صفحه const topic = document.getElementById('topic').value.trim(); const duration = parseInt(document.getElementById('duration').value) || 3; if (!topic) { alert('لطفاً یک موضوع وارد کنید!'); return; } // نمایش وضعیت "در حال ساخت" document.getElementById('loading').style.display = 'block'; document.getElementById('result').style.display = 'none'; document.getElementById('generateBtn').disabled = true; try { // ارسال موضوع به Worker const response = await fetch(WORKER_URL, { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ topic: topic, duration: duration }) }); const data = await response.json(); if (!response.ok || data.error) { throw new Error(data.error || `خطا: ${response.status}`); } // نمایش اسکریپت تولید شده document.getElementById('scriptOutput').textContent = data.script; document.getElementById('providerInfo').textContent = `✅ تولید شده با: ${data.provider} | مدت زمان: ${duration} دقیقه`; document.getElementById('result').style.display = 'block'; // ذخیره اسکریپت برای کپی کردن window.currentScript = data.script; } catch (error) { alert(`❌ خطا: ${error.message}`); console.error('Error:', error); } finally { // پنهان کردن وضعیت "در حال ساخت" document.getElementById('loading').style.display = 'none'; document.getElementById('generateBtn').disabled = false; } }); // دکمه کپی کردن متن document.getElementById('copyBtn').addEventListener('click', () => { const script = document.getElementById('scriptOutput').textContent; navigator.clipboard.writeText(script).then(() => { alert('✅ متن کپی شد!'); }).catch(() => { alert('❌ کپی ناموفق!'); }); });
